// app.js // When the "Search" button is clicked, this function is triggered document.getElementById('searchBtn').addEventListener('click', function () { const query = document.getElementById('searchQuery').value.trim(); // Check if the input is empty if (query === '') { alert('Please enter a search term'); return; } // Call the function to search for books searchBooks(query); }); // Function to fetch books from Google Books API function searchBooks(query) { const apiUrl = `https://www.googleapis.com/books/v1/volumes?q=${query}`; // Google Books API URL // Fetch data from the API fetch(apiUrl) .then(response => response.json()) .then(data => { // Check if there are items in the response if (data.items) { displayResults(data.items); } else { document.getElementById('searchResults').innerHTML = 'No books found.'; } }) .catch(error => { console.error('Error fetching books:', error); document.getElementById('searchResults').innerHTML = 'There was an error with the search.'; }); } // Function to display the books in a customer-friendly format function displayResults(books) { const resultsDiv = document.getElementById('searchResults'); resultsDiv.innerHTML = ''; // Clear any previous results // Loop through the books and display them books.forEach(book => { const bookInfo = book.volumeInfo; const bookElement = document.createElement('div'); bookElement.classList.add('book'); bookElement.innerHTML = `

${bookInfo.title}

Author(s): ${bookInfo.authors ? bookInfo.authors.join(', ') : 'Unknown'}

Published: ${bookInfo.publishedDate || 'N/A'}

Description: ${bookInfo.description || 'No description available.'}

`; resultsDiv.appendChild(bookElement); }); }
